21  package com.bitplan.antlr;

23  import static org.junit.Assert.assertEquals;
24  
25  import org.antlr.v4.runtime.CharStream;
26  import org.antlr.v4.runtime.CommonTokenStream;
27  import org.junit.Test;
28  
29  import com.bitplan.exp.ExpLanguageParser;
30  import com.bitplan.exp.ExpLexer;
31  import com.bitplan.exp.ExpParser;
32  
33  /**
34   * test the expression parser example
35   * 
36   * @author wf
37   *
38   */
39  public class TestExpParser extends BaseTest {
40    
41    @Override
42    public LanguageParser getParser() {
43      return new ExpLanguageParser();
44    }
45  
46    @Test
47    public void testExpressionParser() throws Exception {
48      String expressions[] = { "2*3", "4+5", "(2+3)*(4+5)",
49          // uncomment following line to test gui feature
50          // "(4+5)--(6-7)" 
51      };
52      ExpLanguageParser exprParser = new ExpLanguageParser();
53      for (String expression : expressions) {
54        super.runParser(exprParser, expression, 0);
55      }
56    }
57  
58    @Test
59    public void testExpParser() throws Exception {
60      String expression = "12*(5-6)";
61      CharStream in = LanguageParser.streamForText(expression);
62      ExpLexer lexer = new ExpLexer(in);
63      CommonTokenStream tokens = new CommonTokenStream(lexer);
64      ExpParser parser = new ExpParser(tokens);
65      double result = parser.eval().value;
66      // System.out.println(result);
67      assertEquals(-12.0, result, 0.0001);
68    }
