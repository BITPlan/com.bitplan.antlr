21  package com.bitplan.swingutil;
22  
23  import java.awt.Color;
24  
25  import javax.swing.JScrollPane;
26  import javax.swing.JTextArea;
27  import javax.swing.event.DocumentEvent;
28  import javax.swing.event.DocumentListener;
29  import javax.swing.text.Element;
30  
31  /**
32   * Line Numberer for a JTextArea in a JScrollPane
33   * http://www.javaprogrammingforums.com/java-swing-tutorials/915-how-add-line-numbers-your-jtextarea.html
34   * 
35   * @author wf
36   *
37   */
38  public class LineNumberer implements DocumentListener {
39    JTextArea jta;
40    JTextArea lines;
41    JScrollPane jsp;
42  
43    /**
44     * create a line Numberer for the given scrollpane and text area
45     * @param jsp
46     * @param jta
47     */
48    public LineNumberer(JScrollPane jsp, JTextArea jta) {
49      this.jsp=jsp;
50      this.jta=jta;
51      lines = new JTextArea("00001");
52      
53      lines.setBackground(new Color(220,220,220));
54      lines.setEditable(false);
55      jsp.setRowHeaderView(lines);
56      jsp.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
57      jta.getDocument().addDocumentListener(this);
58    }
59    
60    /**
61     * get the number/stringformat for the given number of lines
62     * @param maxLines
63     * @return
64     */
65    public String getFormat(int maxLines) {
66      // calculate the width e.g. 1 for 8, 2 for 87, 3 for 257 ...
67      int width=1; // for 0-9
68      while (maxLines>=10) {
69        width++; // increment width per 10 longer lines
70        maxLines=maxLines/10; // 
71      }
72      String format=" %0"+width+"d ";   // e.g. " %4d " for 1697
73      return format;
74    }
75    
76    /**
77     * get the line number for the given line
78     * @param line
79     * @return
80     */
81    public String lineNumber(String format,int line) {
82      String lineNumber=String.format(format, line) + System.getProperty("line.separator");
83      return lineNumber;
84    }
85    
86    /**
87     * get the text
88     * @return
89     */
90    public String getText(){
91      int caretPosition = jta.getDocument().getLength();
92      Element root = jta.getDocument().getDefaultRootElement();
93      int maxLine=root.getElementIndex( caretPosition ) + 2;
94      String format=getFormat(maxLine);
95      String text = lineNumber(format,1);
96      for(int i = 2; i < maxLine; i++){
97        text += lineNumber(format,i);
98      }
99      return text;
100   }
101   
102   @Override
103   public void changedUpdate(DocumentEvent de) {
104     lines.setText(getText());
105   }
106 
107   @Override
108   public void insertUpdate(DocumentEvent de) {
109     lines.setText(getText());
110   }
111 
112   @Override
113   public void removeUpdate(DocumentEvent de) {
114     lines.setText(getText());
115   }
116 }
